# Security-engineer-owned regex list. Consumed by
# ch.alpenflight.server.migration.MigrationFolderConventionsTest. Each non-empty,
# non-comment line is a Java regex applied (case-insensitive, multi-line) to
# every *.sql file under db/migration/.
#
# Adding a pattern here means "no migration may contain this." Removing one
# means relaxing a security gate — requires CODEOWNERS sign-off.

# Plain-text credential literal (e.g. `CREATE USER foo WITH PASSWORD 'secret'`).
PASSWORD\s*'

# Privilege-escalation surface. Block role / user creation + grants outright;
# legitimate role provisioning is operator-only and runs outside Flyway.
\bGRANT\s
\bREVOKE\s
\bALTER\s+ROLE\b
\bCREATE\s+USER\b
\bCREATE\s+ROLE\b

# PII data + app-table data committed to source control. Schema-only migrations
# are mandatory; data backfill is S-016 / S-018 territory. Reference tables
# (aircraft_type, location_type, ...) are explicitly NOT in this deny list —
# their canonical UUID seeds ship via V<n>__*.sql by design.
INSERT\s+INTO\s+(person|person_club|audit_logs|audit_log_details|flight|flight_crew|aircraft|aircraft_aircraft_state|aircraft_operating_counter|article|location|inoutbound_point)\b

# Destructive schema ops — allowlist exceptions via PR review.
DROP\s+SCHEMA\b
TRUNCATE\b

# Flyway placeholders — avoid until S-011 tenant-aware tooling lands so they
# don't accidentally inject runtime config into committed SQL.
\$\{[^}]+\}

# ADR 0019 — application owns UUID v7 generation via Hibernate. Migrations
# must not use Postgres-side default ID generation; that would bypass the
# app-generation contract and shift IDs to non-time-ordered v4 quietly.
DEFAULT\s+gen_random_uuid\s*\(
